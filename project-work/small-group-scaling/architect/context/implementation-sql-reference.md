## Task 8: Reduce Non-Raid NPC Stats

```sql
-- Reduce HP to 50% for non-raid NPCs with manual stats
UPDATE npc_types
SET hp = GREATEST(1, ROUND(hp * 0.50))
WHERE level BETWEEN 1 AND 65
  AND raid_target = 0
  AND hp > 0;

-- Reduce max damage to 75%
UPDATE npc_types
SET maxdmg = GREATEST(1, ROUND(maxdmg * 0.75))
WHERE level BETWEEN 1 AND 65
  AND raid_target = 0
  AND maxdmg > 0;

-- Reduce min damage to 65%
UPDATE npc_types
SET mindmg = GREATEST(1, ROUND(mindmg * 0.65))
WHERE level BETWEEN 1 AND 65
  AND raid_target = 0
  AND mindmg > 0;

-- Reduce AC to 82%
UPDATE npc_types
SET AC = GREATEST(1, ROUND(AC * 0.82))
WHERE level BETWEEN 1 AND 65
  AND raid_target = 0
  AND AC > 0;

-- Verify with spot checks:
SELECT id, name, level,
  nt.hp AS new_hp, bk.hp AS orig_hp,
  nt.maxdmg AS new_maxdmg, bk.maxdmg AS orig_maxdmg,
  nt.AC AS new_AC, bk.AC AS orig_AC
FROM npc_types nt
JOIN npc_types_backup_sgs bk ON nt.id = bk.id
WHERE nt.name IN ('Frenzied_Ghoul', 'a_froglok_knight')
```

## Task 14: Generate Rollback SQL Script

```sql
-- Rollback npc_types stats
UPDATE npc_types nt
JOIN npc_types_backup_sgs bk ON nt.id = bk.id
SET nt.hp = bk.hp,
    nt.mindmg = bk.mindmg,
    nt.maxdmg = bk.maxdmg,
    nt.AC = bk.AC;

-- Rollback loottable_entries
UPDATE loottable_entries lte
JOIN loottable_entries_backup_sgs bk
  ON lte.loottable_id = bk.loottable_id AND lte.lootdrop_id = bk.lootdrop_id
SET lte.probability = bk.probability;

-- Rollback lootdrop_entries
UPDATE lootdrop_entries lde
JOIN lootdrop_entries_backup_sgs bk
  ON lde.lootdrop_id = bk.lootdrop_id AND lde.item_id = bk.item_id
SET lde.chance = bk.chance;

-- Rollback spawn2
UPDATE spawn2 s2
JOIN spawn2_backup_sgs bk ON s2.id = bk.id
SET s2.respawntime = bk.respawntime;
```
